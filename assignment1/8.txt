num_list = [23, 56, 12, 67, 36, 89, 34, 50]
count = 0  # Initialize count outside the loop

for index, num in enumerate(num_list):
    if num == 36:
        print("Number found at position:", index)
        break  # Exit the loop when the number is found
    count += 1  # Increment count by 1 in each iteration

print("Total count of numbers in the list:", count)
